function findName(names, index) {
  if (index === names.length) {
    console.log("Name not found");
    return;
  }

  if (names[index] === "Ali") {
    console.log("Found Ali!");
    return;
  }

  findName(names, index + 1);
}

//Best Example For Using Recursion 


function Factorial(n) {
  if (n === 1 || n === 0) return 1; // Base case for it :

  return n * Factorial(n - 1);
}

console.log(Factorial(5));

// Factorial of numbers
function fibonacci(n) {
  let first = 0,
    second = 1;
  process.stdout.write(first + " " + second + " ");
  for (let i = 0; i < n; i++) {
    let third = first + second;
    first = second;
    second = third;
    process.stdout.write(third + " ");
  }
}
fibonacci(5);

function fibonacci(n, first, second) {
  if (n === 0) return; //it  is called the base case for the recurion to solve it

  let third = first + second;

  process.stdout.write(third + " ");

  fibonacci(n - 1, second, third);
}

fibonacci(4, 0, 1);
